Mapped type​

// +? 代表让其变成可选
// -?代表让其变成必选
type AA = {
    name?: string;
    age?: number
}

type BB = {
    [p in keyof AA]-?: AA[p][];
}
type AAA = {
    name: string;
    age: number
}

type BBB = {
    [p in keyof AA]+?: AA[p][];
}
type AAAA = {
    name: string;
    age: number
}

type BBBB = {
    [p in keyof AA]+?: AA[p];
} & {
    age: number
}
let bbbb: BBBB = {
    // name: 'bbbb',
    age: 20
}

if (val2.type === 'add') {
    val2.money
} else if (val2.type === 'remove') {
    val2.name
} else {
    val2
}
