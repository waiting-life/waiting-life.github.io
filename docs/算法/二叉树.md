## 二叉树的层序遍历
只需按层次遍历即可

**递归**

```js
// [1, 2, 4, null, 3]
const root = {
    val: 1,
    left: {
        val: 2,
        left: null,
        right: 4
    },
    right: {
        val: 3
    }
}
const levelOrder = function(root) {
    const result = []
    function _levelorder(node, level) {
        if(!node) return null
        // 当前层数数组初始化
        result[level] = result[level] || []
        result[level].push(node.val)
        // 下一层+1
        _levelorder(node.left, level+1)
        _levelorder(node.right, level+1)
    }
    _levelorder(root, 0)
    return result 
};
levelOrder(root)
// [[1], [2, 4], [3]]
```

## 二叉树的先序遍历
前序遍历：根结点 ---> 左子树 ---> 右子树

**递归**

```js
var preorderTraversal = function(root) {
    const result = []
    // 递归函数
    function _preorder(node) {
        if(!node) return 
        result.push(node.val)
        _preorder(node.left)
        _preorder(node.right)
    }
    _preorder(root)
    return result
};
```

## 二叉树的中序遍历

左子树---> 根结点 ---> 右子树

**递归**

```js
const inorderTraversal = function(root) {
    const result = []
    function _inorder(node) {
        if(!node) return
        _inorder(node.left)
        result.push(node.val)
        _inorder(node.right)
    }
    _inorder(root)
    return result
};
```



## 二叉树的后序遍历

左子树 ---> 右子树 ---> 根结点

**递归**

```js
const postorderTraversal = function(root) {
    const result = []
    function _postorder(node) {
        if(!node) return
        _postorder(node.left)
        _postorder(node.right)
        result.push(node.val)
    }
    _postorder(root)
    return result
};
```
